```vue
<template>
  <div :class="['pricing-card', { featured, 'is-visible': isVisible }]">
    <div :class="['plan-badge', badgeType]">{{ badge }}</div>
    
    <h3 class="plan-name">{{ name }}</h3>
    
    <div class="plan-price">
      <span class="currency">¥</span>
      <span class="amount">{{ displayPrice }}</span>
      <span class="period">/{{ period }}</span>
    </div>
    
    <p v-if="savings" class="savings-text">相比月付 {{ savings }}</p>
    
    <p class="plan-desc">{{ description }}</p>
    
    <ul class="plan-features">
      <li 
        v-for="(feature, i) in features" 
        :key="i"
        :class="{ 
          included: feature.included, 
          excluded: !feature.included,
          highlight: feature.highlight 
        }"
      >
        <span class="check">{{ feature.included ? '✓' : '—' }}</span>
        {{ feature.text }}
      </li>
    </ul>
    
    <router-link 
      :to="ctaLink" 
      :class="['plan-btn', `${badgeType}-btn`]"
    >
      {{ cta }}
    </router-link>
  </div>
</template>

<script setup lang="ts">
import { computed } from 'vue'

interface Feature {
  text: string
  included: boolean
  highlight?: boolean
}

const props = defineProps<{
  name: string
  description: string
  monthlyPrice: number
  annualPrice: number
  badge: string
  badgeType: 'free' | 'pro' | 'annual'
  features: Feature[]
  cta: string
  ctaLink: string
  isAnnual: boolean
  featured?: boolean
  savings?: string | null
  isVisible?: boolean
}>()

const displayPrice = computed(() => {
  const price = props.isAnnual ? props.annualPrice : props.monthlyPrice
  return price === 0 ? '0' : price.toFixed(price < 10 ? 1 : 0)
})

const period = computed(() => props.isAnnual ? '月' : '月')
</script>
```

```sql
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  user_id UUID REFERENCES auth.users(id),
  points INTEGER NOT NULL,
  type TEXT NOT NULL, -- login, post, comment, share, etc.
  description TEXT,
  created_at TIMESTAMP WITH TIME ZONE DEFAULT NOW()
);

-- 创建触发器：自动更新用户等级
CREATE OR REPLACE FUNCTION update_user_level()
RETURNS TRIGGER AS $$
BEGIN
  -- 根据总积分查找对应等级
  SELECT level INTO NEW.current_level
  FROM user_levels
  WHERE NEW.total_points >= min_points AND NEW.total_points <= max_points
  ORDER BY level DESC
  LIMIT 1;
  
  NEW.last_updated = NOW();
  RETURN NEW;
END;
$$ LANGUAGE plpgsql;

CREATE TRIGGER update_level_on_points_change
BEFORE UPDATE ON user_points
FOR EACH ROW
EXECUTE FUNCTION update_user_level();
```

## 五、A/B测试设置

```typescript
// utils/ab-testing.ts

interface Experiment {
  id: string
  name: string
  variants: string[]
  weights: number[]
}

const experiments: Experiment[] = [
  {
    id: 'hero-cta',
    name: 'Hero CTA文案测试',
    variants: ['极速启航', '立即体验', '免费开始', '加入星火'],
    weights: [0.25, 0.25, 0.25, 0.25]
  },
  {
    id: 'pricing-order',
    name: '定价卡片顺序测试',
    variants: ['free-pro-annual', 'pro-free-annual', 'pro-annual-free'],
    weights: [0.34, 0.33, 0.33]
  }
]

// 获取用户的实验分组
export function getVariant(experimentId: string): string {
  const experiment = experiments.find(e => e.id === experimentId)
  if (!experiment) return ''
  
  // 从localStorage获取已分配的分组
  const storageKey = `ab-test-${experimentId}`
  let variant = localStorage.getItem(storageKey)
  
  if (!variant) {
    // 随机分配
    const random = Math.random()
    let cumulativeWeight = 0
    
    for (let i = 0; i < experiment.variants.length; i++) {
      cumulativeWeight += experiment.weights[i]
      if (random <= cumulativeWeight) {
        variant = experiment.variants[i]
        break
      }
    }
    
    localStorage.setItem(storageKey, variant!)
    
    // 发送分配事件
    trackExperiment(experimentId, variant!, 'assigned')
  }
  
  return variant!
}

// 追踪实验事件
function trackExperiment(experimentId: string, variant: string, event: string) {
  if (window.gtag) {
    window.gtag('event', `experiment_${event}`, {
      experiment_id: experimentId,
      variant: variant
    })
  }
  
  // 发送到Supabase
  supabase.from('ab_test_events').insert({
    experiment_id: experimentId,
    variant,
    event,
    timestamp: new Date().toISOString(),
    user_id: getCurrentUserId()
  })
}

// 在组件中使用
const ctaText = computed(() => getVariant('hero-cta'))
```
